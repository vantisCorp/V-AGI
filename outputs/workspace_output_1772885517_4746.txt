                    error="Task validation failed",
                    timestamp=datetime.utcnow()
                )
                
        except Exception as e:
            self.metrics.tasks_failed += 1
            self.task_history[task.id] = {
                "status": "failed",
                "error": str(e),
                "timestamp": datetime.utcnow()
            }
            return AgentResponse(
                agent_id=self.agent_id,
                task_id=task.id,
                status=AgentStatus.FAILED,
                error=str(e),
